cmake_minimum_required(VERSION 4.0)

project(Chess-Engine VERSION 0.2 LANGUAGES CXX)

set(SOURCES
    main.cpp
)

add_executable(chess ${SOURCES})

if(APPLE)
    set_target_properties(chess PROPERTIES
        MACOSX_BUNDLE TRUE
        MACOSX_BUNDLE_GUI_IDENTIFIER com.tommy.chess
        MACOSX_BUNDLE_BUNDLE_NAME "Chess Engine"
        OUTPUT_NAME "Chess Engine"
    )

elseif(WIN32)
    set_target_properties(chess PROPERTIES
        OUTPUT_NAME "ChessEngine"
    )

endif()
